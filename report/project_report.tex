	\section*{سوال ۱: درونیابی لاگرانژ و اسپلاین}
	
	در این بخش، هدف یافتن تابعی است که دقیقاً از مجموعه نقاط داده‌شده شامل
	\[
	(1,1),\ (2,3),\ (3,5),\ (4,8),\ (5,5),\ (6,2)
	\]
	عبور کند. برای این منظور از دو روش درونیابی لاگرانژ و اسپلاین مکعبی
	(\lr{Cubic Spline})
	استفاده شد. همان‌طور که در نتایج خروجی کد پایتون مشخص است، هر دو روش با موفقیت پیاده‌سازی شده و خطای ارزیابی آن‌ها در گره‌های داده‌شده برابر صفر است.
	
	\subsection*{۱. ضابطه توابع به‌دست‌آمده}
	
	\textbf{الف) درونیابی لاگرانژ}
	
	از آنجا که ۶ نقطه داده شده است، چندجمله‌ای درونیاب لاگرانژ یک چندجمله‌ای واحد از درجه ۵ خواهد بود. ضابطه دقیق این تابع پس از محاسبه و ساده‌سازی به فرم زیر به دست آمد:
	
	\begin{equation}
		P(x)=
		0.175x^5
		- 2.958x^4
		+ 18.375x^3
		- 52.041x^2
		+ 68.45x
		- 31.0
	\end{equation}
	
	\textbf{ب) درونیابی اسپلاین مکعبی}
	
	برخلاف روش لاگرانژ، اسپلاین مکعبی از چندجمله‌ای‌های تکه‌ای از درجه ۳ تشکیل شده است که در مرز بازه‌ها، یعنی گره‌ها، دارای مشتقات اول و دوم پیوسته هستند. ضابطه دقیق این توابع برای هر بازه محاسبه شد؛ به عنوان نمونه، ضابطه این تابع در بازه اول $[1,2]$ به صورت زیر است:
	
	\begin{equation}
		S_0(x)=
		-0.186x^3
		+ 0.559x^2
		+ 1.626x
		- 1.0
	\end{equation}
	
	(توابع مربوط به سایر بازه‌ها نیز به طور کامل در خروجی کد درج شده‌اند.)
	
	\subsection*{۲. تحلیل نمودار و مقایسه دو روش}
	
	با توجه به نمودار رسم‌شده در شکل \ref{fig:q1_interpolation}، تفاوت‌های بنیادین رفتار این دو روش به وضوح قابل مشاهده است:
	
	\begin{itemize}
		\item \textbf{رفتار نوسانی لاگرانژ:}
		چندجمله‌ای لاگرانژ، یعنی منحنی پیوسته آبی، به دلیل داشتن درجه بالا، یعنی درجه ۵، تمایل به نوسانات شدیدتری بین نقاط دارد. این موضوع به‌خصوص در بازه انتهایی بین $x=5$ و $x=6$ مشهود است، جایی که منحنی دچار یک افت شدید به سمت پایین شده است. این رفتار نوسانی در لبه‌ها، نمودی از پدیده رونگه
		(\lr{Runge's Phenomenon})
		در درونیابی چندجمله‌ای است.
		
		\item \textbf{پایداری اسپلاین:}
		در مقابل، اسپلاین مکعبی، یعنی خط‌چین قرمز، به دلیل ماهیت تکه‌ای و محدود بودن درجه توابع به عدد ۳، رفتار بسیار نرم‌تر، طبیعی‌تر و پایدارتری از خود نشان می‌دهد. این روش بدون ایجاد نوسانات کاذب بین نقاط، مسیر هموارتری را برای اتصال گره‌ها طی کرده است و برای برازش این نوع داده‌ها انتخاب بهتری محسوب می‌شود.
	\end{itemize}

	\section*{سوال ۲: مصورسازی و تحلیل داده‌های زمان اجرای الگوریتم‌ها}
	
	در این بخش، زمان اجرای سه الگوریتم مختلف
	(\lr{Alg.1, Alg.2, Alg.3})
	بر روی مجموعه‌داده‌هایی با اندازه‌های
	$100\text{KB}$
	الی
	$700\text{KB}$
	مورد ارزیابی قرار گرفته است. داده‌ها به صورت پویا
	(\lr{Dynamic})
	از فایل
	\lr{Excel}
	خوانده شده‌اند؛ به طوری که با اضافه شدن داده جدید، یعنی سطر مربوط به
	$700\text{KB}$،
	کد بدون نیاز به تغییر ساختاری، تمام محاسبات و نمودارها را با موفقیت به‌روزرسانی کرده است.
	
	\subsection*{الف و ب) رسم نمودارها و تحلیل رفتار الگوریتم‌ها}
	
	بر اساس داده‌های استخراج‌شده شامل ۷ رکورد کامل، تحلیل‌های آماری و پیچیدگی زمانی الگوریتم‌ها به شرح زیر است:
	
	\begin{itemize}
		\item \textbf{الگوریتم \lr{Alg.1}:}
		این الگوریتم با اختلاف، بهینه‌ترین عملکرد را دارد. زمان اجرای آن برای داده
		$100\text{KB}$
		برابر
		$50 \text{ ms}$
		است و در بزرگ‌ترین سایز داده، یعنی
		$700\text{KB}$،
		تنها به
		$80 \text{ ms}$
		می‌رسد. رشد زمان اجرای این الگوریتم به ازای هر
		$100\text{KB}$
		تنها
		$5 \text{ ms}$
		است که نشان از مقیاس‌پذیری فوق‌العاده بالا و شیب رشد بسیار کند آن دارد. میانگین کل زمان اجرای این الگوریتم
		$65 \text{ ms}$
		با انحراف معیار بسیار پایین
		$10 \text{ ms}$
		است.
		
		\item \textbf{الگوریتم \lr{Alg.2}:}
		این الگوریتم دارای سربار اولیه
		(\lr{Initial Overhead})
		نسبتاً بالایی است، به طوری که کار خود را از
		$200 \text{ ms}$
		برای داده ۱۰۰ کیلوبایتی آغاز می‌کند. با این حال، شیب رشد آن ملایم است و به ازای هر گام،
		$20 \text{ ms}$
		افزایش می‌یابد. میانگین اجرای این الگوریتم برای کل ۷ رکورد برابر
		$260 \text{ ms}$
		است.
		
		\item \textbf{الگوریتم \lr{Alg.3}:}
		این الگوریتم در سایز کوچک
		$100\text{KB}$
		با زمان
		$100 \text{ ms}$
		عملکردی بهتر از
		\lr{Alg.2}
		دارد، اما نرخ رشد آن بسیار شیب‌دار است و به ازای هر گام،
		$100 \text{ ms}$
		افزایش می‌یابد. این شیب تند باعث می‌شود تا در داده‌های بزرگ، مانند
		$700\text{KB}$،
		زمان اجرای آن به
		$700 \text{ ms}$
		برسد که بدترین عملکرد در میان سه الگوریتم است. میانگین کل این الگوریتم
		$400 \text{ ms}$
		با انحراف معیار بالای
		$200 \text{ ms}$
		ثبت شده است.
	\end{itemize}
	
	نکته آماری مهم این است که به دلیل رشد کاملاً خطی و یکنواخت زمان‌ها در هر سه الگوریتم، مقادیر میانگین
	(\lr{Mean})
	و میانه
	(\lr{Median})
	برای تمامی آن‌ها دقیقاً بر یکدیگر منطبق است.

	\subsection*{ج) محاسبه میانگین زمان اجرای الگوریتم ۲ در بازه ۱۰۰ تا ۶۰۰ کیلوبایت}
	
	طبق خواسته دقیق سوال، در این بخش صرفاً داده‌های مربوط به الگوریتم ۲ برای اندازه‌های
	$100\text{KB}$
	تا
	$600\text{KB}$،
	بدون احتساب سطر جدید ۷۰۰ کیلوبایت، در نظر گرفته شده است.
	
	مقادیر استخراج‌شده از فایل برابرند با:
	\begin{equation*}
		200,\quad 220,\quad 240,\quad 260,\quad 280,\quad 300
	\end{equation*}
	
	محاسبه میانگین
	(\lr{Average})
	به صورت زیر است:
	\begin{equation*}
		\text{Average}
		=
		\frac{200 + 220 + 240 + 260 + 280 + 300}{6}
		=
		\frac{1500}{6}
		=
		250.0 \text{ ms}
	\end{equation*}
	
	این مقدار توسط اسکریپت پایتون نیز به صورت دقیق محاسبه شده و با نتیجه تحلیلی فوق کاملاً مطابقت دارد.
