	\section*{سوال ۱: درونیابی لاگرانژ و اسپلاین}
	
	در این بخش، هدف یافتن تابعی است که دقیقاً از مجموعه نقاط داده‌شده شامل
	\[
	(1,1),\ (2,3),\ (3,5),\ (4,8),\ (5,5),\ (6,2)
	\]
	عبور کند. برای این منظور از دو روش درونیابی لاگرانژ و اسپلاین مکعبی
	(\lr{Cubic Spline})
	استفاده شد. همان‌طور که در نتایج خروجی کد پایتون مشخص است، هر دو روش با موفقیت پیاده‌سازی شده و خطای ارزیابی آن‌ها در گره‌های داده‌شده برابر صفر است.
	
	\subsection*{۱. ضابطه توابع به‌دست‌آمده}
	
	\textbf{الف) درونیابی لاگرانژ}
	
	از آنجا که ۶ نقطه داده شده است، چندجمله‌ای درونیاب لاگرانژ یک چندجمله‌ای واحد از درجه ۵ خواهد بود. ضابطه دقیق این تابع پس از محاسبه و ساده‌سازی به فرم زیر به دست آمد:
	
	\begin{equation}
		P(x)=
		0.175x^5
		- 2.958x^4
		+ 18.375x^3
		- 52.041x^2
		+ 68.45x
		- 31.0
	\end{equation}
	
	\textbf{ب) درونیابی اسپلاین مکعبی}
	
	برخلاف روش لاگرانژ، اسپلاین مکعبی از چندجمله‌ای‌های تکه‌ای از درجه ۳ تشکیل شده است که در مرز بازه‌ها، یعنی گره‌ها، دارای مشتقات اول و دوم پیوسته هستند. ضابطه دقیق این توابع برای هر بازه محاسبه شد؛ به عنوان نمونه، ضابطه این تابع در بازه اول $[1,2]$ به صورت زیر است:
	
	\begin{equation}
		S_0(x)=
		-0.186x^3
		+ 0.559x^2
		+ 1.626x
		- 1.0
	\end{equation}
	
	(توابع مربوط به سایر بازه‌ها نیز به طور کامل در خروجی کد درج شده‌اند.)
	
	\subsection*{۲. تحلیل نمودار و مقایسه دو روش}
	
	با توجه به نمودار رسم‌شده در شکل \ref{fig:q1_interpolation}، تفاوت‌های بنیادین رفتار این دو روش به وضوح قابل مشاهده است:
	
	\begin{itemize}
		\item \textbf{رفتار نوسانی لاگرانژ:}
		چندجمله‌ای لاگرانژ، یعنی منحنی پیوسته آبی، به دلیل داشتن درجه بالا، یعنی درجه ۵، تمایل به نوسانات شدیدتری بین نقاط دارد. این موضوع به‌خصوص در بازه انتهایی بین $x=5$ و $x=6$ مشهود است، جایی که منحنی دچار یک افت شدید به سمت پایین شده است. این رفتار نوسانی در لبه‌ها، نمودی از پدیده رونگه
		(\lr{Runge's Phenomenon})
		در درونیابی چندجمله‌ای است.
		
		\item \textbf{پایداری اسپلاین:}
		در مقابل، اسپلاین مکعبی، یعنی خط‌چین قرمز، به دلیل ماهیت تکه‌ای و محدود بودن درجه توابع به عدد ۳، رفتار بسیار نرم‌تر، طبیعی‌تر و پایدارتری از خود نشان می‌دهد. این روش بدون ایجاد نوسانات کاذب بین نقاط، مسیر هموارتری را برای اتصال گره‌ها طی کرده است و برای برازش این نوع داده‌ها انتخاب بهتری محسوب می‌شود.
	\end{itemize}

	\subsection*{ج) محاسبه میانگین زمان اجرای الگوریتم ۲ در بازه ۱۰۰ تا ۶۰۰ کیلوبایت}

	\section*{سوال ۳: محاسبه عددی انتگرال و مقایسه روش‌ها}
	
	در این بخش، هدف محاسبه مقدار انتگرال معین تابع
	$f(x)=e^{x^2}$
	در بازه
	$[0,1]$
	است. این انتگرال دارای پاسخ تحلیلی با توابع مقدماتی نیست و باید از طریق روش‌های عددی تقریب زده شود. مقدار مرجع و بسیار دقیق این انتگرال برابر با
	$1.462651745907$
	محاسبه شده است که از آن برای ارزیابی خطای مطلق روش‌های مختلف استفاده می‌کنیم.
	
	\subsection*{الف) بررسی روش‌های نیوتن-کوتس، ذوزنقه‌ای و سیمپسون}
	
	روش‌های ذوزنقه‌ای
	(\lr{Trapezoidal})
	و سیمپسون
	(\lr{Simpson})
	از دسته فرمول‌های نیوتن-کوتس هستند که بازه انتگرال‌گیری را به
	$n$
	زیربازه مساوی با گام
	$h$
	تقسیم می‌کنند.
	
	\begin{itemize}
		\item \textbf{روش ذوزنقه‌ای مرکب:}
		این روش تابع را با خطوط راست، یعنی چندجمله‌ای درجه ۱، تقریب می‌زند. از نظر تئوری، خطای برشی این روش از مرتبه
		$O(h^2)$
		است. داده‌های خروجی این موضوع را کاملاً اثبات می‌کنند؛ به عنوان مثال، وقتی تعداد بازه‌ها
		($n$)
		از ۱۶ به ۳۲، یعنی ۲ برابر، افزایش می‌یابد، خطای مطلق از
		$1.77 \times 10^{-3}$
		به
		$4.42 \times 10^{-4}$
		کاهش می‌یابد. نسبت این دو خطا تقریباً برابر ۴ است
		($2^2=4$)
		که تاییدکننده مرتبه خطای تقریب است. این روش در
		$n=128$
		به خطای
		$2.77 \times 10^{-5}$
		رسیده است که برای کارهای دقیق، نیازمند محاسبات سنگین‌تری است.
		
		\item \textbf{روش سیمپسون مرکب:}
		این روش از سهمی‌ها، یعنی چندجمله‌ای درجه ۲، برای تقریب استفاده می‌کند و خطای تئوریک آن از مرتبه
		$O(h^4)$
		است. با بررسی داده‌ها می‌بینیم با دو برابر شدن
		$n$
		از ۱۶ به ۳۲، خطا از
		$4.58 \times 10^{-6}$
		به
		$2.88 \times 10^{-7}$
		می‌رسد. نسبت این کاهش خطا تقریباً برابر ۱۶ است
		($2^4=16$).
		این روش به دلیل مرتبه همگرایی بالاتر، با همان تعداد گره
		$n=128$،
		به خطای بسیار ناچیز
		$1.13 \times 10^{-9}$
		دست یافته است که عملکردی بسیار بهتر از روش ذوزنقه‌ای دارد.
	\end{itemize}
	
	\subsection*{ب) بررسی انتگرال‌گیری گوسی، گاوس-لژاندر}
	
	برخلاف روش‌های نیوتن-کوتس که از نقاط با فواصل یکسان استفاده می‌کنند، روش گاوس-لژاندر
	(\lr{Gauss-Legendre})
	هم نقاط، یعنی گره‌ها، و هم وزن‌ها را به گونه‌ای بهینه می‌کند که بالاترین دقت ممکن به دست آید. یک فرمول
	$n$
	نقطه‌ای گوسی می‌تواند چندجمله‌ای‌هایی تا درجه
	$2n-1$
	را به طور دقیق انتگرال‌گیری کند.
	
	نتایج اجرای این روش شگفت‌انگیز است. از آنجا که تابع
	$f(x)=e^{x^2}$
	تابعی بی‌نهایت بار مشتق‌پذیر و هموار
	(\lr{Smooth})
	است، روش گوسی اصطلاحاً دارای «همگرایی طیفی»
	(\lr{Spectral Convergence})
	برای آن است.
	
	همان‌طور که در خروجی‌ها پیداست، این روش تنها با استفاده از
	\textbf{۸ نقطه ارزیابی}
	به خطای
	$1.02 \times 10^{-13}$
	رسیده است که در حد دقت ماشین
	(\lr{Machine Precision})
	است. این در حالی است که روش‌های قبلی حتی با ۱۲۸ نقطه ارزیابی نتوانستند به این دقت نزدیک شوند.
	
	\subsection*{پ) تحلیل نمودارها و نتیجه‌گیری نهایی}
	
	برای درک بهتر رفتار الگوریتم‌ها، دو نمودار رسم شده است. این نمودارها در شکل
	\ref{fig:integration_plots}
	نمایش داده شده‌اند.
	
	\begin{itemize}
		\item \textbf{نمودار سمت چپ:}
		مساحت زیر نمودار تابع
		$f(x)=e^{x^2}$
		را نشان می‌دهد که تحدب رو به بالای آن مشهود است.
		
		\item \textbf{نمودار سمت راست، نمودار همگرایی:}
		این نمودار که در مقیاس لگاریتمی رسم شده است، حیاتی‌ترین بخش تحلیل است. محور افقی تعداد نقاط یا زیربازه‌ها و محور عمودی خطای مطلق است. شیب هر خط نشان‌دهنده سرعت همگرایی آن روش است. خط آبی، یعنی ذوزنقه، شیب ملایمی دارد. خط بنفش، یعنی سیمپسون، شیب تندتری دارد. اما خط نارنجی، یعنی گاوس-لژاندر، سقوطی تقریباً عمودی را تجربه می‌کند که نشان می‌دهد با افزودن تنها چند نقطه، خطا به صورت نمایی
		(\lr{Exponential})
		کاهش می‌یابد.
	\end{itemize}

	(\lr{Most Efficient})
	روش است؛ زیرا با کمترین تعداد فراخوانی تابع، یعنی کمترین هزینه محاسباتی، بیشترین دقت را به ارمغان می‌آورد. با این حال، اگر داده‌ها به صورت جدولی با فواصل یکسان داده شده باشند، نه به صورت تابع تحلیلی، استفاده از روش سیمپسون بهترین گزینه در دسترس خواهد بود.
