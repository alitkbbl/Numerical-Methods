	در این بخش، هدف یافتن تابعی است که دقیقاً از مجموعه نقاط داده شده شامل
	$(1,1), (2,3), (3,5), (4,8), (5,5), (6,2)$
	عبور کند. برای این منظور از دو روش درونیابی لاگرانژ و اسپلاین مکعبی
	(\lr{Cubic Spline})
	استفاده شد. همان‌طور که در نتایج خروجی کد پایتون مشخص است، هر دو روش با موفقیت پیاده‌سازی شده و خطای ارزیابی آن‌ها در گره‌های داده شده برابر صفر است.
	
	\section*{۱. ضابطه توابع به دست آمده}
	
	\textbf{الف) درونیابی لاگرانژ:}
	
	از آنجا که ۶ نقطه داده شده است، چندجمله‌ای درونیاب لاگرانژ یک چندجمله‌ای واحد از درجه ۵ خواهد بود. ضابطه دقیق این تابع پس از محاسبه و ساده‌سازی به فرم زیر به دست آمد:
	
	\begin{equation}
		P(x) =
		0.175x^5
		- 2.958x^4
		+ 18.375x^3
		- 52.041x^2
		+ 68.45x
		- 31.0
	\end{equation}
	
	\textbf{ب) درونیابی اسپلاین مکعبی:}
	
	برخلاف روش لاگرانژ، اسپلاین مکعبی از چندجمله‌ای‌های تکه‌ای از درجه ۳ تشکیل شده است که در مرز بازه‌ها، یعنی گره‌ها، دارای مشتقات اول و دوم پیوسته هستند. ضابطه دقیق این توابع برای هر بازه محاسبه شد؛ به عنوان نمونه، ضابطه این تابع در بازه اول $[1, 2]$ به صورت زیر است:
	
	\begin{equation}
		S_0(x) =
		-0.186x^3
		+ 0.559x^2
		+ 1.626x
		- 1.0
	\end{equation}
	
	(توابع مربوط به سایر بازه‌ها نیز به طور کامل در خروجی کد درج شده‌اند).
	
	\section*{۲. تحلیل نمودار و مقایسه دو روش}
	
	با توجه به نمودار رسم شده، شکل \ref{fig:q1_interpolation}، تفاوت‌های بنیادین رفتار این دو روش به وضوح قابل مشاهده است:
	
	\begin{itemize}
		\item \textbf{رفتار نوسانی لاگرانژ:}
		چندجمله‌ای لاگرانژ، یعنی منحنی پیوسته آبی، به دلیل داشتن درجه بالا، درجه ۵، تمایل به نوسانات شدیدتری بین نقاط دارد. این موضوع به خصوص در بازه انتهایی بین $x=5$ و $x=6$ مشهود است، جایی که منحنی دچار یک افت شدید به سمت پایین شده است. این رفتار نوسانی در لبه‌ها، نمودی از پدیده رونگه
		(\lr{Runge's Phenomenon})
		در درونیابی چندجمله‌ای است.
		
		\item \textbf{پایداری اسپلاین:}
		در مقابل، اسپلاین مکعبی، یعنی خط‌چین قرمز، به دلیل ماهیت تکه‌ای و محدود بودن درجه توابع به عدد ۳، رفتار بسیار نرم‌تر، طبیعی‌تر و پایدارتری از خود نشان می‌دهد. این روش بدون ایجاد نوسانات کاذب بین نقاط، مسیر هموارتری را برای اتصال گره‌ها طی کرده است و برای برازش این نوع داده‌ها انتخاب بهتری محسوب می‌شود.
	\end{itemize}

	\subsection*{۱. تحلیل آماری و رفتار الگوریتم‌ها}
	
	بر اساس نتایج استخراج‌شده، می‌توان رفتار هر یک از الگوریتم‌ها را از منظر پیچیدگی زمانی و مقیاس‌پذیری
	(\lr{Scalability})
	ارزیابی کرد:
	
	\begin{itemize}
		\item \textbf{الگوریتم \lr{Alg.1}:}
		این الگوریتم بهترین عملکرد را از خود نشان می‌دهد. زمان اجرای آن برای داده
		$100\text{KB}$
		برابر
		$50 \text{ ms}$
		بوده و با شیب بسیار ملایمی رشد می‌کند. میانگین زمان اجرای آن نیز
		$62.5 \text{ ms}$
		است. این امر نشان‌دهنده مقیاس‌پذیری عالی این الگوریتم برای داده‌های بزرگ‌تر است.
		
		\item \textbf{الگوریتم \lr{Alg.2}:}
		این الگوریتم دارای سربار اولیه
		(\lr{Initial Overhead})
		نسبتاً بالایی است و از
		$200 \text{ ms}$
		شروع می‌شود؛ با این حال، شیب رشد آن کند است. میانگین زمان اجرای این الگوریتم در بازه بررسی‌شده دقیقاً برابر
		$250.0 \text{ ms}$
		محاسبه شد.
		
		\item \textbf{الگوریتم \lr{Alg.3}:}
		اگرچه این الگوریتم در حجم داده‌های کم عملکرد بهتری نسبت به
		\lr{Alg.2}
		دارد، اما به دلیل داشتن شیب رشد تند، به سرعت دچار افت عملکرد می‌شود و برای داده‌های بزرگ، مانند
		$600\text{KB}$
		با زمان
		$600 \text{ ms}$،
		بدترین انتخاب ممکن است.
	\end{itemize}
	
	نکته قابل توجه دیگر این است که به دلیل رشد کاملاً خطی و فواصل منظم داده‌ها، در هر سه الگوریتم، مقدار میانگین
	(\lr{Mean})
	دقیقاً با مقدار میانه
	(\lr{Median})
	برابر است.
