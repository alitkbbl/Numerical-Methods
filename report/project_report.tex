	\section*{سوال ۱: درونیابی لاگرانژ و اسپلاین}
	
	در این بخش، هدف این است که برای مجموعه نقاط زیر، تابعی پیدا کنیم که از همه آن‌ها عبور کند:
	\[
	(1,1),\ (2,3),\ (3,5),\ (4,8),\ (5,5),\ (6,2)
	\]
	برای این کار از دو روش درونیابی لاگرانژ و اسپلاین مکعبی
	(\lr{Cubic Spline})
	استفاده شده است. خروجی برنامه نشان می‌دهد که هر دو روش درست پیاده‌سازی شده‌اند، چون مقدار خطا در گره‌های داده‌شده برابر صفر است و هر دو تابع دقیقاً از نقاط اصلی عبور می‌کنند.
	
	\subsection*{نحوه پیاده‌سازی}
	
	برای درونیابی این شش نقطه، دو روش اصلی بررسی شد:
	
	\textbf{۱. چندجمله‌ای لاگرانژ:}
	در این روش از تابع
	\lr{scipy.interpolate.lagrange}
	استفاده شد. این تابع یک چندجمله‌ای یکتا می‌سازد که از تمام نقاط داده‌شده عبور می‌کند. چون در روش لاگرانژ فقط یک چندجمله‌ای برای کل بازه ساخته می‌شود، خروجی نهایی یک تابع سراسری و هموار است؛ البته همین ویژگی گاهی باعث ایجاد نوسان‌های ناخواسته، مخصوصاً در نزدیکی انتهای بازه، می‌شود.
	
	\textbf{۲. اسپلاین مکعبی طبیعی:}
	برای این بخش از کلاس
	\lr{CubicSpline}
	استفاده شد. در اسپلاین، به جای ساختن یک چندجمله‌ای واحد برای کل نقاط، بین هر دو نقطه متوالی یک چندجمله‌ای مکعبی جداگانه تعریف می‌شود. به همین دلیل، تابع نهایی به صورت تکه‌ای ساخته می‌شود و معمولاً رفتار طبیعی‌تر و کنترل‌شده‌تری بین نقاط دارد.
	
	\textbf{اعتبارسنجی نتایج:}
	برای اطمینان از درست بودن پیاده‌سازی، مقدار هر دو تابع در نقاط اصلی داده بررسی شد. نتیجه نشان داد که هر دو روش دقیقاً از نقاط داده عبور می‌کنند. همچنین با کمک
	\lr{sympy}
	فرم نمادین توابع نیز استخراج شد تا علاوه بر نمودار، نمایش تحلیلی آن‌ها هم در دسترس باشد.
	
	\textbf{نمایش و مقایسه:}
	در پایان، منحنی هر دو روش همراه با نقاط اصلی رسم شد. این مقایسه کمک می‌کند تفاوت رفتار دو روش بهتر دیده شود: لاگرانژ یک چندجمله‌ای یکتا و سراسری می‌سازد، در حالی که اسپلاین با ساختار تکه‌ای خود معمولاً نوسان کمتری دارد و در بسیاری از داده‌های واقعی انتخاب مطمئن‌تری است.
	
	\subsection*{۱. ضابطه توابع به‌دست‌آمده}
	
	\textbf{الف) درونیابی لاگرانژ}
	
	از آنجا که ۶ نقطه داده شده است، چندجمله‌ای لاگرانژ از درجه ۵ خواهد بود. پس از محاسبه و ساده‌سازی، ضابطه چندجمله‌ای به شکل زیر به دست آمد:
	
	\begin{equation}
		P(x)=
		0.175x^5
		- 2.958x^4
		+ 18.375x^3
		- 52.041x^2
		+ 68.45x
		- 31.0
	\end{equation}
	
	\textbf{ب) درونیابی اسپلاین مکعبی}
	
	در روش اسپلاین مکعبی، تابع نهایی از چندجمله‌ای‌های درجه ۳ تشکیل می‌شود. این چندجمله‌ای‌ها در گره‌ها به هم متصل می‌شوند و علاوه بر مقدار تابع، مشتق اول و دوم آن‌ها نیز در نقاط اتصال پیوسته است. به عنوان نمونه، ضابطه تابع در بازه اول، یعنی
	$[1,2]$،
	به صورت زیر محاسبه شد:
	
	\begin{equation}
		S_0(x)=
		-0.186x^3
		+ 0.559x^2
		+ 1.626x
		- 1.0
	\end{equation}
	
	ضابطه سایر بازه‌ها نیز به طور کامل در خروجی برنامه درج شده است.
	
	\subsection*{۲. تحلیل نمودار و مقایسه دو روش}
	
	با توجه به نمودار شکل
	\ref{fig:q1_interpolation}،
	تفاوت رفتار دو روش کاملاً مشخص است:
	
	\begin{itemize}
		\item \textbf{رفتار لاگرانژ:}
		منحنی لاگرانژ، که با رنگ آبی نمایش داده شده، یک چندجمله‌ای درجه ۵ است. این منحنی از همه نقاط عبور می‌کند، اما به دلیل سراسری بودن چندجمله‌ای، در بعضی قسمت‌ها دچار نوسان بیشتری می‌شود. این مسئله به‌ویژه در بازه انتهایی بین
		$x=5$
		و
		$x=6$
		بیشتر دیده می‌شود. چنین رفتاری نمونه‌ای از پدیده رونگه
		(\lr{Runge's Phenomenon})
		در درونیابی چندجمله‌ای است.
		
		\item \textbf{رفتار اسپلاین:}
		در مقابل، اسپلاین مکعبی که با خط‌چین قرمز نشان داده شده، مسیر نرم‌تر و طبیعی‌تری بین نقاط طی می‌کند. چون این روش به صورت تکه‌ای عمل می‌کند و در هر بازه فقط از یک چندجمله‌ای درجه ۳ استفاده می‌شود، نوسان‌های کاذب کمتری ایجاد می‌کند. به همین دلیل، برای چنین داده‌هایی معمولاً اسپلاین انتخاب مناسب‌تری است.
	\end{itemize}

		\caption{مقایسه درونیابی لاگرانژ و اسپلاین مکعبی روی نقاط داده‌شده}

	\end{figure}

	\section*{سوال ۲: مصورسازی و تحلیل زمان اجرای الگوریتم‌ها}
	
	در این بخش، زمان اجرای سه الگوریتم
	(\lr{Alg.1, Alg.2, Alg.3})
	برای مجموعه‌داده‌هایی با اندازه
	$100\,\text{KB}$
	تا
	$700\,\text{KB}$
	بررسی شده است. داده‌ها از فایل اکسل خوانده شده‌اند و ساختار کد به صورتی نوشته شده که با اضافه شدن داده جدید، مانند سطر مربوط به
	$700\,\text{KB}$،
	بدون نیاز به تغییر جدی در برنامه، محاسبات و نمودارها به‌روزرسانی شوند.
	
	\subsection*{نحوه پیاده‌سازی}
	
	در این سوال، داده‌های مربوط به زمان اجرای الگوریتم‌ها مستقیماً از فایل اکسل خوانده شدند. برای این کار از تابع
	\lr{pandas.read\_excel}
	استفاده شد تا اطلاعات به شکل منظم وارد برنامه شوند. خواندن سطرها و ستون‌ها نیز به صورت پویا انجام شد تا کد در برابر اضافه شدن داده‌های جدید انعطاف‌پذیر باشد.
	
	\textbf{مصورسازی داده‌ها:}
	برای اینکه رفتار الگوریتم‌ها بهتر قابل مقایسه باشد، سه نوع نمودار تولید شد:
	
	\begin{itemize}
		\item \textbf{نمودار میله‌ای:} برای مقایسه مستقیم زمان اجرای الگوریتم‌ها در اندازه‌های مختلف داده
		\item \textbf{نمودار خطی:} برای مشاهده روند رشد زمان اجرا و بررسی رفتار کلی هر الگوریتم
		\item \textbf{نمودار جعبه‌ای:} برای بررسی پراکندگی، پایداری و تفاوت آماری زمان اجرای الگوریتم‌ها
	\end{itemize}
	
	\textbf{محاسبات آماری:}
	برای هر الگوریتم، شاخص‌های آماری اصلی شامل میانگین، میانه و انحراف معیار محاسبه شد. همچنین میانگین زمان اجرای الگوریتم ۲ در بازه
	$100\,\text{KB}$
	تا
	$600\,\text{KB}$
	به صورت جداگانه به دست آمد. این محاسبه با فیلتر کردن
	\lr{DataFrame}
	بر اساس اندازه داده انجام شد.
	
	در مجموع، کد به گونه‌ای طراحی شده که هم در بخش پردازش داده و هم در بخش رسم نمودارها، وابستگی کمی به تعداد رکوردها داشته باشد و بتواند با ورودی‌های جدید نیز به‌خوبی کار کند.
	
	\subsection*{الف و ب) رسم نمودارها و تحلیل رفتار الگوریتم‌ها}
	
	با توجه به ۷ رکورد موجود در داده‌ها، رفتار سه الگوریتم به صورت زیر قابل تحلیل است:
	
	\begin{itemize}
		\item \textbf{الگوریتم \lr{Alg.1}:}
		این الگوریتم بهترین عملکرد را در میان سه الگوریتم دارد. زمان اجرای آن برای داده
		$100\,\text{KB}$
		برابر
		$50\,\text{ms}$
		است و حتی برای بزرگ‌ترین ورودی، یعنی
		$700\,\text{KB}$،
		فقط به
		$80\,\text{ms}$
		می‌رسد. رشد زمان اجرا در این الگوریتم بسیار آرام است و به ازای هر
		$100\,\text{KB}$
		تقریباً
		$5\,\text{ms}$
		افزایش پیدا می‌کند. میانگین زمان اجرای آن
		$65\,\text{ms}$
		و انحراف معیار آن
		$10\,\text{ms}$
		است که نشان‌دهنده پایداری بسیار خوب الگوریتم است.
		
		\item \textbf{الگوریتم \lr{Alg.2}:}
		این الگوریتم در شروع، زمان اجرای بیشتری نسبت به دو الگوریتم دیگر دارد و برای داده
		$100\,\text{KB}$
		از
		$200\,\text{ms}$
		شروع می‌شود. با این حال، شیب رشد آن نسبتاً ملایم است و در هر گام فقط
		$20\,\text{ms}$
		افزایش دارد. میانگین زمان اجرای این الگوریتم برای کل داده‌ها برابر
		$260\,\text{ms}$
		است. بنابراین، اگرچه شروع آن کندتر است، اما در داده‌های بزرگ‌تر رفتار قابل قبول‌تری نسبت به الگوریتم سوم دارد.
		
		\item \textbf{الگوریتم \lr{Alg.3}:}
		این الگوریتم برای داده کوچک
		$100\,\text{KB}$
		با زمان
		$100\,\text{ms}$
		عملکرد بهتری نسبت به
		\lr{Alg.2}
		دارد، اما مشکل اصلی آن نرخ رشد زیاد زمان اجراست. زمان اجرای آن در هر گام حدود
		$100\,\text{ms}$
		افزایش می‌یابد و در داده
		$700\,\text{KB}$
		به
		$700\,\text{ms}$
		می‌رسد. میانگین زمان اجرای آن
		$400\,\text{ms}$
		و انحراف معیار آن
		$200\,\text{ms}$
		است که نشان می‌دهد این الگوریتم نسبت به افزایش حجم داده حساسیت زیادی دارد.
	\end{itemize}
	
	یک نکته آماری جالب این است که چون زمان اجرای هر سه الگوریتم روندی کاملاً خطی و منظم دارد، مقدار میانگین
	(\lr{Mean})
	و میانه
	(\lr{Median})
	برای هر الگوریتم با هم برابر شده است.

	\subsection*{ج) محاسبه میانگین زمان اجرای الگوریتم ۲ در بازه ۱۰۰ تا ۶۰۰ کیلوبایت}
	
	در این بخش فقط داده‌های مربوط به الگوریتم ۲ برای اندازه‌های
	$100\,\text{KB}$
	تا
	$600\,\text{KB}$
	در نظر گرفته شده است؛ بنابراین سطر مربوط به
	$700\,\text{KB}$
	در این محاسبه وارد نمی‌شود.
	
	مقادیر استخراج‌شده از فایل برابرند با:
	\begin{equation*}
		200,\quad 220,\quad 240,\quad 260,\quad 280,\quad 300
	\end{equation*}
	
	میانگین این مقادیر به صورت زیر محاسبه می‌شود:
	\begin{equation*}
		\text{Average}
		=
		\frac{200 + 220 + 240 + 260 + 280 + 300}{6}
		=
		\frac{1500}{6}
		=
		250.0\,\text{ms}
	\end{equation*}
	
	این مقدار در خروجی برنامه پایتون نیز دقیقاً برابر
	$250.0\,\text{ms}$
	به دست آمده و با محاسبه دستی کاملاً مطابقت دارد.

	\section*{سوال ۳: محاسبه عددی انتگرال و مقایسه روش‌ها}
	
	در این بخش، هدف محاسبه مقدار انتگرال معین تابع

	در بازه
	$[0,1]$
	است. این انتگرال جواب ساده‌ای بر حسب توابع مقدماتی ندارد؛ بنابراین باید مقدار آن را با روش‌های عددی تقریب بزنیم. مقدار مرجع انتگرال برابر است با:
	\[
	1.462651745907
	\]
	این مقدار به عنوان مبنای مقایسه در نظر گرفته شده و خطای مطلق هر روش نسبت به آن محاسبه شده است.
	
	\subsection*{نحوه پیاده‌سازی}
	
	برای محاسبه عددی انتگرال
	\[
	f(x)=e^{x^2}
	\]
	در بازه
	\[
	[0,1]
	\]
	از سه روش استفاده شد:
	
	\textbf{۱. روش ذوزنقه‌ای مرکب:}
	در این روش، بازه انتگرال‌گیری به
	$n$
	زیربازه مساوی تقسیم شد و مقدار انتگرال روی هر زیربازه با تقریب ذوزنقه‌ای به دست آمد. این روش پیاده‌سازی ساده‌ای دارد، اما در مقایسه با روش‌های دقیق‌تر، برای رسیدن به خطای کم به تعداد زیربازه بیشتری نیاز دارد.
	
	\textbf{۲. روش سیمپسون ۱/۳ مرکب:}
	در روش سیمپسون، به جای تقریب خطی، از تقریب سهمی‌وار استفاده می‌شود. این روش فقط زمانی قابل استفاده است که تعداد زیربازه‌ها زوج باشد. دقت سیمپسون معمولاً از روش ذوزنقه‌ای بیشتر است و با تعداد تقسیم‌بندی کمتر می‌تواند جواب دقیق‌تری تولید کند.
	
	\textbf{۳. روش گاوس-لژاندر:}
	در روش گاوس-لژاندر، نقاط نمونه‌برداری و وزن‌ها به صورت بهینه انتخاب می‌شوند. ابتدا گره‌ها و وزن‌ها روی بازه استاندارد محاسبه شده و سپس به بازه
	$[0,1]$
	تبدیل می‌شوند. مزیت مهم این روش این است که با تعداد کمی ارزیابی تابع، می‌تواند به دقت بسیار بالایی برسد.
	
	\textbf{مقدار مرجع و مقایسه خطا:}
	چون این انتگرال پاسخ بسته ساده‌ای ندارد، برای محاسبه از
	\lr{scipy.integrate.quad}
	استفاده شد. سپس خروجی هر سه روش با مقدار مرجع مقایسه و خطای مطلق آن‌ها محاسبه شد.
	
	\textbf{تحلیل همگرایی:}
	نتایج نشان می‌دهد که روش گاوس-لژاندر با تعداد ارزیابی کمتر، دقیق‌ترین نتیجه را تولید می‌کند. همچنین روش سیمپسون نسبت به روش ذوزنقه‌ای سریع‌تر همگرا می‌شود؛ زیرا مرتبه خطای آن
	\[
	O(h^4)
	\]
	است، در حالی که مرتبه خطای روش ذوزنقه‌ای برابر است با:
	\[
	O(h^2)
	\]
	
	\subsection*{الف) بررسی روش‌های نیوتن-کوتس، ذوزنقه‌ای و سیمپسون}
	
	روش‌های ذوزنقه‌ای
	(\lr{Trapezoidal})
	و سیمپسون
	(\lr{Simpson})
	از خانواده فرمول‌های نیوتن-کوتس هستند. در این روش‌ها، بازه انتگرال‌گیری به
	$n$
	زیربازه مساوی با طول گام
	$h$
	تقسیم می‌شود.
	
	\begin{itemize}
		\item \textbf{روش ذوزنقه‌ای مرکب:}
		در این روش، تابع با خط‌های راست، یعنی چندجمله‌ای‌های درجه ۱، تقریب زده می‌شود. خطای نظری این روش از مرتبه
		$O(h^2)$
		است. نتایج عددی نیز همین موضوع را نشان می‌دهند. برای مثال، وقتی تعداد زیربازه‌ها از
		$n=16$
		به
		$n=32$
		می‌رسد، خطای مطلق از
		$1.77 \times 10^{-3}$
		به
		$4.42 \times 10^{-4}$
		کاهش پیدا می‌کند. نسبت این دو خطا تقریباً ۴ است و با رابطه
		$2^2=4$
		هماهنگی دارد. در
		$n=128$
		خطا به
		$2.77 \times 10^{-5}$
		می‌رسد، اما همچنان برای رسیدن به دقت‌های خیلی بالا نیاز به زیربازه‌های بیشتری دارد.
		
		\item \textbf{روش سیمپسون مرکب:}
		در روش سیمپسون، تابع با سهمی‌ها، یعنی چندجمله‌ای‌های درجه ۲، تقریب زده می‌شود. خطای نظری این روش از مرتبه
		$O(h^4)$
		است. طبق نتایج، با دو برابر شدن
		$n$
		از ۱۶ به ۳۲، خطا از
		$4.58 \times 10^{-6}$
		به
		$2.88 \times 10^{-7}$
		می‌رسد. نسبت کاهش خطا تقریباً ۱۶ است که با رابطه
		$2^4=16$
		مطابقت دارد. این روش با همان
		$n=128$
		به خطای بسیار کوچک
		$1.13 \times 10^{-9}$
		رسیده و از روش ذوزنقه‌ای به‌مراتب دقیق‌تر است.
	\end{itemize}
	
	\subsection*{ب) بررسی روش گاوس-لژاندر}
	
	در روش‌های نیوتن-کوتس، نقاط نمونه‌برداری فاصله‌های مساوی دارند؛ اما در روش گاوس-لژاندر
	(\lr{Gauss-Legendre})
	هم محل نقاط و هم وزن‌ها به شکل بهینه انتخاب می‌شوند. یک فرمول گوسی
	$n$
	نقطه‌ای می‌تواند انتگرال چندجمله‌ای‌هایی تا درجه
	$2n-1$
	را دقیق محاسبه کند.
	
	نتایج این روش بسیار قابل توجه است. تابع
	$f(x)=e^{x^2}$
	تابعی هموار
	(\lr{Smooth})
	است و به همین دلیل روش گاوس-لژاندر برای آن بسیار سریع همگرا می‌شود. در واقع، این روش فقط با
	\textbf{۸ نقطه ارزیابی}
	به خطای
	$1.02 \times 10^{-13}$
	رسیده است؛ خطایی که عملاً در محدوده دقت ماشین
	(\lr{Machine Precision})
	قرار دارد.
	
	این نتیجه نشان می‌دهد که برای توابع هموار، روش گاوس-لژاندر می‌تواند با هزینه محاسباتی بسیار کمتر، دقتی بسیار بالاتر از روش‌های ذوزنقه‌ای و سیمپسون ارائه دهد.
	
	\subsection*{پ) تحلیل نمودارها و نتیجه‌گیری}
	
	برای مقایسه بهتر روش‌ها، دو نمودار در شکل
	\ref{fig:integration_plots}
	رسم شده است.
	
	\begin{itemize}
		\item \textbf{نمودار سمت چپ:}
		این نمودار تابع
		$f(x)=e^{x^2}$
		و مساحت زیر منحنی آن را در بازه
		$[0,1]$
		نشان می‌دهد. از شکل تابع مشخص است که نمودار در این بازه صعودی و محدب است.
		
		\item \textbf{نمودار سمت راست:}
		این نمودار روند کاهش خطای مطلق روش‌ها را در مقیاس لگاریتمی نمایش می‌دهد. محور افقی تعداد نقاط یا زیربازه‌ها و محور عمودی خطای مطلق است. خط مربوط به روش ذوزنقه‌ای کاهش آرام‌تری دارد، خط سیمپسون با شیب بیشتری پایین می‌آید، اما روش گاوس-لژاندر افت بسیار سریع‌تری را تجربه می‌کند. این رفتار نشان می‌دهد که در این مسئله، با اضافه شدن تعداد کمی نقطه، خطای روش گوسی به سرعت کاهش می‌یابد.
	\end{itemize}
